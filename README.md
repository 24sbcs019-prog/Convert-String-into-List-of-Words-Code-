# Convert-String-into-List-of-Words-Code-
str1 = "The quick brown fox jumps over the lazy dog."
print(str1.split(' '))

str1 = "The-quick-brown-fox-jumps-over-the-lazy-dog."
print(str1.split('-'))

Output:
['The', 'quick', 'brown', 'fox', 'jumps', 'over', 'the', 'lazy', 'dog.']
['The', 'quick', 'brown', 'fox', 'jumps', 'over', 'the', 'lazy', 'dog.']
